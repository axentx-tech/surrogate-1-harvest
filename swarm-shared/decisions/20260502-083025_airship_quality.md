import requests
import time
import json
import os

def fetch_files(repo_id, path):
    # Use list_repo_tree with recursive=False
    response = requests.get(f"https://huggingface.co/api/repo/tree/{repo_id}/{path}", params={"recursive": False})
    files = response.json()["files"]
    # Use CDN downloads
    file_urls = [f"https://huggingface.co/{repo_id}/resolve/main/{file['path']}" for file in files]
    return file_urls

def download_file(url):
    response = requests.get(url)
    return response.content

def download_dataset_files(file_paths):
    for file_path in file_paths:
        file_url = f"https://huggingface.co/datasets/{file_path}/resolve/main/{file_path}"
        # Download the file using the HF CDN
        os.system(f"wget {file_url}")

def main():
    repo_id = "your-repo-id"
    path = "your-path"
    try:
        file_urls = fetch_files(repo_id, path)
        file_paths = [url.split("/")[-1] for url in file_urls]
        # Save the list of file paths to a JSON file
        with open("file_paths.json", "w") as f:
            json.dump(file_paths, f)
        # Download the dataset files using the HF CDN
        download_dataset_files(file_paths)
    except requests.exceptions.HTTPError as e:
        if e.response.status_code == 429:
            print("Rate limit exceeded. Waiting 360s before retrying.")
            time.sleep(360)

if __name__ == "__main__":
    main()
